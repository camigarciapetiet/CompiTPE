.DATA
aux@programa DD ?
diseno@programa DW ?
recfuncion db "none", 0
@aux1 DW ?
@aux2 DW ?
@aux3 DW ?
@aux4 DW ?
@aux5 DW ?
@aux6 DW ?

.CODE
EXIT:
 invoke ExitProcess, 0
DIV_CERO:
 invoke MessageBox, NULL, addr DIV_CERO, NULL, MB_OK
 JMP EXIT
OVERFLOW_ERROR:
 invoke MessageBox, NULL, addr OVERFLOW_ERROR, NULL, MB_OK
 JMP EXIT
ERROR_REC_MUTUA:
 invoke MessageBox, NULL, addr ERROR_REC_MUTUA, NULL, MB_OK
 JMP EXIT
programa:
MOV BX, 400
CMP BX, -999
JO OVERFLOW_ERROR
MOV AX, -999
ADD AX, 400
MOV @aux1, AX
MOV AX, @aux1
MOV diseno@programa, AX
MOV BX, 0
CMP BX, 0
JE DIV_CERO
MOV @aux2, 0
MOV AX, diseno@programa
IDIV @aux2
MOV @aux3, AX
MOV AX, @aux3
MOV diseno@programa, AX
MOV BX, 5
CMP BX, 5
JO OVERFLOW_ERROR
MOV AX, 5
ADD AX, 5
MOV @aux4, AX
MOV AX, 5
IMUL AX, 5
MOV @aux5, AX
MOV BX, @aux5
CMP BX, @aux4
JO OVERFLOW_ERROR
MOV AX, @aux4
ADD AX, @aux5
MOV @aux6, AX
MOV AX, @aux6
MOV diseno@programa, AX
END programa
invoke MessageBox, NULL, addr FIN, NULL, MB_OK
